// Select elements
const el = document.querySelector("#myId");    // first match

// Change content & style
el.textContent = "Hello!";

btn.addEventListener("click", myFunction);

querySelectorAll → ALL matches

🟩 Module 2 — Variables & Strings

// Variables
let name = "Melika";   // can reassign
const age = 25;       // cannot reassign

// String methods
name.length            // 6 — no ()
name.toUpperCase()     // "MELIKA"
name.toLowerCase()     // "melika"
name.replace("a","o") // "Meliko"
name.slice(0, 3)       // "Mel"
name.indexOf("M")      // 0 (position)

// Type conversion
Number("42")   // string → number
String(42)    // number → string

// Strict equality
5 === 5       // true (value + type)
5 == "5"     // true (value only)
⚠️ Remember! .length → NO parentheses (it's a property!)
.replace(old, new) → always this order

🟪 Module 3 — Arrays

// Create array
let fruits = ["apple", "banana", "mango"];

// Access items (index starts at 0!)
fruits[0]      // "apple"
fruits.length  // 3

// Add / Remove
fruits.push("peach")  // add END → returns new LENGTH
fruits.pop()          // remove END → returns ITEM
fruits.unshift("kiwi") // add FRONT → returns new LENGTH
fruits.shift()        // remove FRONT → returns ITEM

// Convert
fruits.join(", ")     // array → string
"a,b,c".split(",")  // string → array

// Modify middle
fruits.splice(1, 1, "grape"); // at[1], remove 1, add "grape"
⚠️ The Big Trick! push() & unshift() → return new LENGTH
pop() & shift() → return the removed ITEM

🟧 Module 4 — Conditionals & Loops

// if / else if / else
if (score >= 90) {
  console.log("A");
} else if (score >= 80) {
  console.log("B");
} else {
  console.log("C");
}

// Ternary
let msg = age >= 18 ? "adult" : "minor";

// Switch
switch(day) {
  case "Mon": console.log("Monday"); break;
  default: console.log("other day");
}

// for loop
for (let i = 0; i < arr.length; i++) {
  console.log(arr[i]);
}

// while loop — manual i++ required!
let i = 0;
while (i < arr.length) {
  console.log(arr[i]);
  i++; // MUST DO THIS!
}

// forEach
arr.forEach((item) => {
  console.log(item); // no i++ needed
});
⚠️ Remember! break → EXIT loop entirely
continue → SKIP current iteration
while → YOU must write i++ manually!

🟥 Module 5 — Functions & Scope

// Named function
function greet(name) {
  return "Hello, " + name;
}
let msg = greet("Melika"); // "Hello, Melika"

// Anonymous function
const greet = function(name) {
  return "Hello, " + name;
};

// Arrow function
const greet = (name) => {
  return "Hello, " + name;
};

// Scope
let global = "everywhere"; // global ✅

function myFunc() {
  let local = "only here"; // local
  console.log(global); // ✅ can access global
}
console.log(local); // ❌ ReferenceError!
⚠️ Remember! return is OPTIONAL
local variables → only inside their function
global variables → accessible everywhere

🟫 Module 6 — Events

// Basic event listener
btn.addEventListener("click", myFunction);
// NO () on function name!

// Event object
btn.addEventListener("click", (event) => {
  console.log(event.target);      // element clicked
  console.log(event.target.nodeName); // "BUTTON", "IMG"
});

// preventDefault
form.addEventListener("submit", (event) => {
  event.preventDefault(); // stop form submitting
});

// Event delegation — one listener on parent
thumbBar.addEventListener("click", (event) => {
  if (event.target.nodeName === "IMG") {
    displayImage(event.target.src);
  }
});

// Remove event listener
btn.removeEventListener("click", myFunction);
⚠️ Remember! nodeName returns UPPERCASE: "IMG" not "img"
event delegation = ONE listener on parent
preventDefault = stops default browser action
